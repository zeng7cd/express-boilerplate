### 3. 导入方式对比

#### 3.1 导入路径统一

**重构前 (❌ 混乱的导入方式):**

```typescript
// 文件: src/modules/users/services/user.service.ts
import { prisma } from '../../../core/database';  // ❌ 相对路径
import { env } from '../../../core/config/config/envConfig';  // ❌ 相对路径
import type { Prisma } from '../../../generated/prisma/client';  // ❌ 相对路径

// 文件: src/modules/auth/services/auth.service.ts
import { prisma } from '@/core/database';  // ✅ 绝对路径
import { env } from '@/core/config/config/envConfig';  // ❌ 路径冗余
```

**重构后 (✅ 统一的导入方式):**

```typescript
// 文件: src/modules/users/services/user.service.ts
import { prisma } from '@/core/database';  // ✅ 统一绝对路径
import { env } from '@/core/config';  // ✅ 简化路径
import type { Prisma } from '@/generated/prisma/client';  // ✅ 统一绝对路径

// 文件: src/modules/auth/services/auth.service.ts
import { prisma } from '@/core/database';  // ✅ 统一绝对路径
import { env } from '@/core/config';  // ✅ 简化路径

// 或者使用统一入口 (推荐)
import { prisma, env } from '@/core';  // ✅ 最简洁方式
```
