##### 1.1 JWT 认证实现

**步骤 1: 安装依赖**

```bash
pnpm add jsonwebtoken bcryptjs
pnpm add -D @types/jsonwebtoken @types/bcryptjs
```

**步骤 2: 创建认证中间件**

```typescript
// src/middleware/auth.ts
import jwt from 'jsonwebtoken';
import type { Request, Response, NextFunction } from 'express';
import { env } from '@/config/envConfig';

interface JWTPayload {
  userId: string;
  email: string;
  role: string;
}

export interface AuthenticatedRequest extends Request {
  user?: JWTPayload;
}

export const authenticateToken = (req: AuthenticatedRequest, res: Response, next: NextFunction) => {
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1];

  if (!token) {
    return res.status(401).json({
      success: false,
      message: 'Access token required',
    });
  }

  jwt.verify(token, env.JWT_SECRET, (err, decoded) => {
    if (err) {
      return res.status(403).json({
        success: false,
        message: 'Invalid or expired token',
      });
    }

    req.user = decoded as JWTPayload;
    next();
  });
};
```

**步骤 3: 创建授权中间件**

```typescript
// src/middleware/authorization.ts
import type { Response, NextFunction } from 'express';
import type { AuthenticatedRequest } from './auth';

export const authorize = (roles: string[]) => {
  return (req: AuthenticatedRequest, res: Response, next: NextFunction) => {
    if (!req.user) {
      return res.status(401).json({
        success: false,
        message: 'Authentication required',
      });
    }

    if (!roles.includes(req.user.role)) {
      return res.status(403).json({
        success: false,
        message: 'Insufficient permissions',
      });
    }

    next();
  };
};
```

**步骤 2: API 密钥验证中间件**

```typescript
// src/middleware/apiKey.ts
import crypto from 'crypto';
import { prisma } from '@/database';

export const validateApiKey = async (req: Request, res: Response, next: NextFunction) => {
  const apiKey = req.headers['x-api-key'] as string;

  if (!apiKey) {
    return res.status(401).json({
      success: false,
      message: 'API key required',
    });
  }

  const keyHash = crypto.createHash('sha256').update(apiKey).digest('hex');

  const apiKeyRecord = await prisma.apiKey.findUnique({
    where: { keyHash, isActive: true },
  });

  if (!apiKeyRecord || (apiKeyRecord.expiresAt && apiKeyRecord.expiresAt < new Date())) {
    return res.status(401).json({
      success: false,
      message: 'Invalid or expired API key',
    });
  }

  // 更新最后使用时间
  await prisma.apiKey.update({
    where: { id: apiKeyRecord.id },
    data: { lastUsedAt: new Date() },
  });

  next();
};
```

##### 2.1 统一日志接口设计

// src/utils/logger/types.ts

**步骤 2: 实现统一日志器**

```typescript
// src/utils/logger/logger.ts
import pino from 'pino';
import type { ILogger, LoggerConfig } from './types';

class UnifiedLogger implements ILogger {
  private pinoLogger: pino.Logger;

  constructor(config: LoggerConfig) {
    this.pinoLogger = pino({
      level: config.level,
      transport: config.pretty
        ? {
            target: 'pino-pretty',
            options: {
              colorize: true,
              translateTime: 'SYS:standard',
            },
          }
        : undefined,
    });
  }

  debug(message: string, meta?: Record<string, any>): void {
    this.pinoLogger.debug(meta, message);
  }

  info(message: string, meta?: Record<string, any>): void {
    this.pinoLogger.info(meta, message);
  }

  warn(message: string, meta?: Record<string, any>): void {
    this.pinoLogger.warn(meta, message);
  }

  error(message: string, error?: Error, meta?: Record<string, any>): void {
    this.pinoLogger.error({ err: error, ...meta }, message);
  }

  child(bindings: Record<string, any>): ILogger {
    const childLogger = this.pinoLogger.child(bindings);
    return new UnifiedLogger({ level: 'info', pretty: false }) as any;
  }
}
```

**步骤 3: 创建日志器工厂**

```typescript
// src/utils/logger/factory.ts
import { env } from '@/config/envConfig';
import { UnifiedLogger } from './logger';
import type { LoggerConfig } from './types';

export class LoggerFactory {
  private static instance: LoggerFactory;
  private appLogger: UnifiedLogger;

  private constructor() {
    const config: LoggerConfig = {
      level: env.LOGGER_LEVEL,
      pretty: env.isDevelopment,
      file: {
        enabled: env.LOG_TO_FILE,
        path: './logs/app.log',
        maxSize: env.LOG_FILE_MAX_SIZE,
        maxFiles: env.LOG_FILE_MAX_FILES,
      },
    };

    this.appLogger = new UnifiedLogger(config);
  }

  static getInstance(): LoggerFactory {
    if (!LoggerFactory.instance) {
      LoggerFactory.instance = new LoggerFactory();
    }
    return LoggerFactory.instance;
  }

  getAppLogger(): UnifiedLogger {
    return this.appLogger;
  }

  createModuleLogger(moduleName: string): UnifiedLogger {
    return this.appLogger.child({ module: moduleName }) as UnifiedLogger;
  }
}

export const logger = LoggerFactory.getInstance().getAppLogger();
```

**实施时间**: 2-3 天
**预期收益**: 简化维护，提升性能，统一日志格式

**步骤 3: 集成监控中间件**

```typescript
// src/middleware/monitoring.ts
import type { Request, Response, NextFunction } from 'express';
import { httpRequestDuration, httpRequestTotal } from '@/monitoring/metrics';

export const monitoringMiddleware = (req: Request, res: Response, next: NextFunction) => {
  const start = Date.now();

  res.on('finish', () => {
    const duration = (Date.now() - start) / 1000;
    const route = req.route?.path || req.path;
    const method = req.method;
    const statusCode = res.statusCode.toString();

    httpRequestDuration.labels(method, route, statusCode).observe(duration);

    httpRequestTotal.labels(method, route, statusCode).inc();
  });

  next();
};
```

##### 5.2 查询优化

```typescript
// src/api/largeScreen/repositories/vehicle.repo.ts
export class VehicleRepository {
  // 批量查询优化
  async findManyWithPagination(params: { page: number; limit: number; status?: number }) {
    const { page, limit, status } = params;
    const skip = (page - 1) * limit;

    const [vehicles, total] = await Promise.all([
      prisma.vehicle.findMany({
        where: status !== undefined ? { currentStatus: status } : undefined,
        skip,
        take: limit,
        orderBy: { createdTime: 'desc' },
        include: {
          _count: {
            select: { detectors: true, events: true },
          },
        },
      }),
      prisma.vehicle.count({
        where: status !== undefined ? { currentStatus: status } : undefined,
      }),
    ]);

    return {
      data: vehicles,
      pagination: {
        page,
        limit,
        total,
        totalPages: Math.ceil(total / limit),
      },
    };
  }

  // 使用事务进行批量操作
  async batchUpdateStatus(updates: Array<{ id: string; status: number }>) {
    return prisma.$transaction(
      updates.map((update) =>
        prisma.vehicle.update({
          where: { vehicleId: update.id },
          data: { currentStatus: update.status },
        }),
      ),
    );
  }
}
```
